import scrapy
from urllib.parse import urljoin

class ProductSpider(scrapy.Spider):
    name = 'product'
    allowed_domains = ['target.example.com']
    start_urls = ['https://target.example.com/category/electronics']

    def parse(self, response):
        # и§ЈжһҗеҲ—иЎЁйЎөпјҢжҸҗеҸ–е•Ҷе“Ғй“ҫжҺҘ
        items = response.css('div.product-item')
        for item in items:
            product = {
                'product_id': item.css('::attr(data-id)').get(),
                'title': item.css('h3.title::text').get(),
                'price': float(item.css('span.price::text').get('0')),
                'url': urljoin(response.url, item.css('a::attr(href)').get())
            }
            yield scrapy.Request(
                product['url'],
                callback=self.parse_detail,
                meta={'item': product}
            )
        # зҝ»йЎө
        next_page = response.css('a.next-page::attr(href)').get()
        if next_page:
            yield scrapy.Request(urljoin(response.url, next_page))

    def generate_token(self, product_id):
        import hashlib
        import time
        ts = str(int(time.time() * 1000))
        raw = f"{product_id}_{ts}_secret_key"
        return hashlib.md5(raw.encode()).hexdigest()

from Crypto.Cipher import AES
from Crypto.Util.Padding import unpad
import base64
import time

def decrypt_field(encrypted_b64, ts):
    """и§ЈеҜҶAjaxе“Қеә”дёӯзҡ„еҠ еҜҶеӯ—ж®ө"""
    key = f"key_{ts}".encode('utf-8').ljust(16, b'0')[:16]

        AES.block_size
    )
    return decrypted.decode('utf-8')

8.3.2 Downloader Middleware дёӯй—ҙд»¶жҺҘе…Ҙ Cookie жұ вҖӢ

е…Ҳе®ҡд№ү Cookie жұ зҡ„ж•°жҚ®з»“жһ„гҖӮCookie жұ жң¬иҙЁдёҠжҳҜдёҖдёӘеӯҳеӮЁеӨҡдёӘжңүж•Ҳ Cookie зҡ„йӣҶеҗҲпјҢйңҖиҰҒж”ҜжҢҒд»ҘдёӢж“ҚдҪңпјҡиҺ·еҸ–дёҖдёӘеҸҜз”Ё CookieгҖҒж·»еҠ ж–° CookieгҖҒж Үи®°еӨұж•Ҳ CookieгҖҒз»ҹи®Ў Cookie зҠ¶жҖҒгҖӮ

import redis
import json
import time

class CookiePool:
    def __init__(self, host='localhost', port=6379):
        self.redis = redis.Redis(host=host, port=port, db=2)
        self.pool_key = 'cookie_pool'
        self.blocked_key = 'cookie_blocked'

    def get_cookie(self):
        """йҡҸжңәиҺ·еҸ–дёҖдёӘеҸҜз”ЁCookie"""
        cookies = self.redis.lrange(self.pool_key, 0, -1)
        if not cookies:
            return None
        cookie = random.choice(cookies)
        return json.loads(cookie)

    def add_cookie(self, cookie_data):
        """ж·»еҠ ж–°CookieеҲ°жұ дёӯ"""
        cookie_data['add_time'] = time.time()
        self.redis.rpush(self.pool_key, json.dumps(cookie_data))

    def block_cookie(self, cookie_id):
        """ж Үи®°CookieеӨұж•ҲпјҢд»Һжұ дёӯз§»йҷӨ"""
        cookies = self.redis.lrange(self.pool_key, 0, -1)
        for cookie in cookies:
            data = json.loads(cookie)
            if data.get('id') == cookie_id:
                self.redis.lrem(self.pool_key, 1, cookie)
                data['block_time'] = time.time()
                self.redis.lpush(self.blocked_key, json.dumps(data))
                break

иҝҷйҮҢз”Ё Redis дҪңдёә Cookie жұ зҡ„еӯҳеӮЁеҗҺз«ҜпјҢеҺҹеӣ жңүдёӨдёӘпјҡдёҖжҳҜ Redis зҡ„еҲ—иЎЁз»“жһ„еӨ©з„¶йҖӮеҗҲеҒҡжұ еӯҗпјҢlpush/rpop ж“ҚдҪңжҳҜеҺҹеӯҗзҡ„пјӣдәҢжҳҜеҗҺз»ӯиҰҒжү©еұ•дёәеҲҶеёғејҸзҲ¬иҷ«ж—¶пјҢеӨҡдёӘ Worker иҠӮзӮ№йңҖиҰҒе…ұдә«еҗҢдёҖдёӘ Cookie жұ пјҢRedis еӨ©з„¶ж”ҜжҢҒеӨҡиҝӣзЁӢи®ҝй—®гҖӮ

жҺҘдёӢжқҘжҳҜ Downloader Middleware зҡ„е®һзҺ°пјҡ

import random
import logging

class CookieRotationMiddleware:
    def __init__(self, cookie_pool):
        self.cookie_pool = cookie_pool
        self.logger = logging.getLogger(__name__)

    def process_request(self, request, spider):
        # еҸӘеҜ№йңҖиҰҒCookieзҡ„иҜ·жұӮж·»еҠ Cookie
        if request.meta.get('need_cookie', False):
            cookie = self.cookie_pool.get_cookie()
            if cookie:
                request.cookies = cookie.get('cookies')
                request.meta['cookie_id'] = cookie.get('id')
                self.logger.debug(f"дҪҝз”ЁCookie: {cookie.get('id')}")
            else:
                self.logger.warning("Cookieжұ дёәз©әпјҒ")
                return None
        return None

    def process_response(self, request, response, spider):
        # жЈҖжөӢCookieжҳҜеҗҰеӨұж•Ҳ
        if response.status == 401:
            cookie_id = request.meta.get('cookie_id')
            if cookie_id:
                self.cookie_pool.block_cookie(cookie_id)
                self.logger.info(f"Cookie {cookie_id} е·ІеӨұж•ҲпјҢд»Һжұ дёӯз§»йҷӨ")
                return request  # иҝ”еӣһrequestи§ҰеҸ‘йҮҚиҜ•
        return response

иҝҷж®өд»Јз ҒжңүдёӨдёӘж ёеҝғйҖ»иҫ‘гҖӮprocess_request ж–№жі•еңЁиҜ·жұӮеҸ‘еҮәд№ӢеүҚжү§иЎҢпјҢд»Һ Cookie жұ дёӯйҡҸжңәеҸ–дёҖдёӘ Cookie йҷ„еҠ еҲ°иҜ·жұӮдёҠгҖӮжіЁж„ҸиҝҷйҮҢз”ЁдәҶ request.meta['need_cookie'] жқҘжҺ§еҲ¶е“ӘдәӣиҜ·жұӮйңҖиҰҒеҠ  CookieвҖ”вҖ”еҲ—иЎЁйЎөдёҚйңҖиҰҒзҷ»еҪ•жҖҒпјҢеҸӘжңүиҜҰжғ…йЎөзҡ„ Ajax жҺҘеҸЈжүҚйңҖиҰҒгҖӮ

8.3.3 Cookie иҪ®жҚўзӯ–з•ҘвҖӢ

Cookie иҪ®жҚўзңӢиө·жқҘз®ҖеҚ•вҖ”вҖ”жҜҸдёӘиҜ·жұӮеҲҶй…ҚдёҖдёӘдёҚеҗҢзҡ„ Cookie е°ұиЎҢгҖӮдҪҶе®һйҷ…ж“ҚдҪңдёӯпјҢиҪ®жҚўзӯ–з•Ҙзҡ„еҘҪеқҸзӣҙжҺҘеҪұе“ҚжҲҗеҠҹзҺҮе’Ң Cookie ж¶ҲиҖ—йҖҹеәҰгҖӮжҖ•жөӘзҢ«жөӢиҜ•иҝҮдёүз§ҚиҪ®жҚўзӯ–з•ҘпјҢеҗ„жңүдјҳеҠЈгҖӮ

йҡҸжңәиҪ®жҚўпјҡжҜҸж¬ЎиҜ·жұӮд»Һжұ дёӯйҡҸжңәеҸ–дёҖдёӘ CookieгҖӮдјҳзӮ№жҳҜе®һзҺ°з®ҖеҚ•пјҢCookie дҪҝз”ЁеқҮеҢҖгҖӮзјәзӮ№жҳҜеҗҢдёҖдёӘ Cookie еҸҜиғҪеңЁзҹӯж—¶й—ҙеҶ…иў«еӨҡж¬ЎдҪҝз”ЁпјҢеҰӮжһңдёӨж¬ЎиҜ·жұӮй—ҙйҡ”еӨӘзҹӯпјҢеҸҜиғҪи§ҰеҸ‘йў‘зҺҮйҷҗеҲ¶гҖӮ

# йЎәеәҸиҪ®жҚўпјҲRedisеҺҹеӯҗйҖ’еўһпјү

жқғйҮҚиҪ®жҚўпјҡж №жҚ® Cookie зҡ„еҺҶеҸІжҲҗеҠҹзҺҮеҲҶй…ҚжқғйҮҚпјҢжҲҗеҠҹзҺҮй«ҳзҡ„ Cookie иў«йҖүдёӯзҡ„жҰӮзҺҮжӣҙеӨ§гҖӮдјҳзӮ№жҳҜиғҪиҮӘеҠЁдјҳе…ҲдҪҝз”Ёй«ҳиҙЁйҮҸ CookieгҖӮзјәзӮ№жҳҜе®һзҺ°еӨҚжқӮпјҢйңҖиҰҒз»ҙжҠӨжҜҸдёӘ Cookie зҡ„жҲҗеҠҹ/еӨұиҙҘи®Ўж•°гҖӮ

# жқғйҮҚиҪ®жҚў
def get_weighted_cookie(self):
    cookies = self.redis.lrange('cookie_pool', 0, -1)
    weights = []
    for c in cookies:
        data = json.loads(c)
        success = data.get('success_count', 1)
        fail = data.get('fail_count', 0)
        weight = success / (success + fail + 1)
        weights.append(weight)
    return json.choices(cookies, weights=weights)[0]

е®һжөӢдёӢжқҘпјҢеҜ№дәҺжҲ‘д»¬зҡ„зӣ®ж ҮзҪ‘з«ҷпјҢйЎәеәҸиҪ®жҚўзҡ„ж•ҲжһңжңҖеҘҪгҖӮеҺҹеӣ жҳҜзӣ®ж ҮзҪ‘з«ҷзҡ„ IP йў‘зҺҮйҷҗеҲ¶жҳҜжҢү CookieпјҲеҚіиҙҰеҸ·пјүз»ҙеәҰи®Ўж•°зҡ„пјҢйЎәеәҸиҪ®жҚўиғҪзЎ®дҝқжҜҸдёӘ Cookie зҡ„иҜ·жұӮй—ҙйҡ”жңҖеқҮеҢҖгҖӮйҡҸжңәиҪ®жҚўеҒ¶е°”дјҡеҮәзҺ°иҝһз»ӯдҪҝз”ЁеҗҢдёҖ Cookie зҡ„жғ…еҶөпјҢеҜјиҮҙйў‘зҺҮйҷҗеҲ¶гҖӮжқғйҮҚиҪ®жҚўеңЁ Cookie иҙЁйҮҸе·®ејӮдёҚеӨ§ж—¶пјҢж•Ҳжһңе’ҢйЎәеәҸиҪ®жҚўе·®дёҚеӨҡпјҢдҪҶе®һзҺ°жӣҙеӨҚжқӮгҖӮ

иҪ®жҚўзӯ–з•ҘжІЎжңүй“¶еј№пјҢжңҖйҖӮеҗҲдҪ зӣ®ж ҮзҪ‘з«ҷзҡ„зӯ–з•ҘпјҢеҸӘиғҪйҖҡиҝҮе®һжөӢжқҘзЎ®е®ҡгҖӮ

8.3.4 Cookie еӨұж•ҲиҮӘеҠЁж Үи®°дёҺжӣҝжҚўвҖӢ

Cookie еӨұж•ҲжҳҜеҝ…з„¶еҸ‘з”ҹзҡ„дәӢжғ…гҖӮе…ій”®дёҚжҳҜеҰӮдҪ•йҒҝе…ҚеӨұж•ҲпјҢиҖҢжҳҜеҰӮдҪ•еҝ«йҖҹжЈҖжөӢеӨұж•Ҳе№¶иҮӘеҠЁжӣҝжҚўгҖӮжҖ•жөӘзҢ«и®ҫи®ЎдәҶдёҖеҘ—е®Ңж•ҙзҡ„ Cookie з”ҹе‘Ҫе‘Ёжңҹз®ЎзҗҶжңәеҲ¶гҖӮ

Cookie зҡ„з”ҹе‘Ҫе‘ЁжңҹеҲҶдёәеӣӣдёӘзҠ¶жҖҒпјҡжҙ»и·ғпјҲactiveпјүгҖҒеҶ·еҚҙпјҲcoolingпјүгҖҒеӨұж•ҲпјҲblockedпјүгҖҒиҝҮжңҹпјҲexpiredпјүгҖӮжҙ»и·ғзҠ¶жҖҒзҡ„ Cookie еҸҜд»ҘжӯЈеёёдҪҝз”ЁгҖӮеҶ·еҚҙзҠ¶жҖҒиЎЁзӨәиҜҘ Cookie еҲҡеҲҡи§ҰеҸ‘дәҶдёҖж¬Ўй”ҷиҜҜпјҲжҜ”еҰӮ 401пјүпјҢжҡӮж—¶дёҚдҪҝз”ЁпјҢзӯүеҫ…дёҖж®өж—¶й—ҙеҗҺиҮӘеҠЁжҒўеӨҚдёәжҙ»и·ғзҠ¶жҖҒгҖӮеӨұж•ҲзҠ¶жҖҒиЎЁзӨәиҜҘ Cookie е·ІзЎ®и®Өж— жі•дҪҝз”ЁпјҲжҜ”еҰӮиҝһз»ӯи§ҰеҸ‘ 401 дёүж¬ЎпјүпјҢйңҖиҰҒд»Һжұ дёӯз§»йҷӨгҖӮиҝҮжңҹзҠ¶жҖҒиЎЁзӨә Cookie и¶…иҝҮжңҖеӨ§дҪҝз”Ёж—¶й—ҙпјҢйңҖиҰҒйҮҚж–°зҷ»еҪ•иҺ·еҸ–гҖӮ

        self.max_fail_count = 3      # жңҖеӨ§еӨұиҙҘж¬Ўж•°
        self.max_age_hours = 2        # жңҖеӨ§еӯҳжҙ»ж—¶й—ҙпјҲе°Ҹж—¶пјү
        self.cooling_seconds = 60     # еҶ·еҚҙж—¶й—ҙпјҲз§’пјү

    def use_cookie(self, cookie_id):
        """ж Үи®°Cookieиў«дҪҝз”Ё"""
        self.redis.hincrby('cookie_stats', f'{cookie_id}:used', 1)

    def report_fail(self, cookie_id):
        """жҠҘе‘ҠCookieдҪҝз”ЁеӨұиҙҘ"""
        fail_count = self.redis.hincrby(
            'cookie_stats', f'{cookie_id}:fail', 1
        )
        if fail_count >= self.max_fail_count:
            self.block_cookie(cookie_id)
            return 'blocked'
        else:
            self.cool_cookie(cookie_id)
            return 'cooling'

    def report_success(self, cookie_id):
        """жҠҘе‘ҠCookieдҪҝз”ЁжҲҗеҠҹпјҢйҮҚзҪ®еӨұиҙҘи®Ўж•°"""
        self.redis.hset('cookie_stats', f'{cookie_id}:fail', 0)
        self.redis.hincrby('cookie_stats', f'{cookie_id}:success', 1)

    def clean_expired(self):
        """жё…зҗҶиҝҮжңҹCookie"""
        now = time.time()
        cookies = self.redis.lrange('cookie_pool', 0, -1)
        for c in cookies:
            data = json.loads(c)
            age = now - data.get('add_time', 0)
            if age > self.max_age_hours * 3600:
                self.block_cookie(data['id'])

class DoubleAntiAntiCrawlMiddleware:
    def __init__(self, cookie_pool, proxy_pool):
        self.cookie_pool = cookie_pool
        self.proxy_pool = proxy_pool

    def process_request(self, request, spider):
        # и®ҫзҪ®д»ЈзҗҶ
        proxy = self.proxy_pool.get_proxy()
        if proxy:
            request.meta['proxy'] = proxy['url']
            request.meta['proxy_id'] = proxy['id']

        # и®ҫзҪ®Cookie
        if request.meta.get('need_cookie'):
            cookie = self.cookie_pool.get_cookie()
            if cookie:
                request.cookies = cookie['cookies']
                request.meta['cookie_id'] = cookie['id']
        return None

    def process_response(self, request, response, spider):
        if response.status == 401:
            cookie_id = request.meta.get('cookie_id')
            if cookie_id:
                self.cookie_pool.block_cookie(cookie_id)
            return request  # и§ҰеҸ‘йҮҚиҜ•
        if response.status in (403, 429):
            proxy_id = request.meta.get('proxy_id')
            if proxy_id:
                self.proxy_pool.block_proxy(proxy_id)
            return request  # и§ҰеҸ‘йҮҚиҜ•
        return response

еҸҢйҮҚиҪ®жҚўзҡ„ж•ҲжһңжҳҜжҳҫи‘—зҡ„гҖӮжҲ‘еҒҡиҝҮеҜ№жҜ”жөӢиҜ•пјҡеҚ•з”Ёд»ЈзҗҶжұ ж—¶жҲҗеҠҹзҺҮдёә 78%пјҢеҚ•з”Ё Cookie жұ ж—¶жҲҗеҠҹзҺҮдёә 71%пјҢдёӨиҖ…й…ҚеҗҲдҪҝз”Ёж—¶жҲҗеҠҹзҺҮжҸҗеҚҮеҲ° 94%гҖӮеҺҹеӣ еҫҲз®ҖеҚ•пјҡеҸҢйҮҚиҪ®жҚўи®©жҜҸдёӘиҜ·жұӮзҡ„"жҢҮзә№"йғҪдёҚеҗҢпјҢжңҚеҠЎз«Ҝзҡ„йЈҺжҺ§зі»з»ҹж— жі•е°ҶеӨҡдёӘиҜ·жұӮе…іиҒ”еҲ°еҗҢдёҖдёӘзҲ¬иҷ«е®һдҪ“гҖӮ

from scrapy_redis.spiders import RedisSpider

class DistributedProductSpider(RedisSpider):
    name = 'distributed_product'
    redis_key = 'product:start_urls'

    def parse(self, response):
        # и§ЈжһҗйҖ»иҫ‘дёҺеҚ•жңәзүҲе®Ңе…ЁдёҖиҮҙ
        items = response.css('div.product-item')
        for item in items:
            product = {
                'product_id': item.css('::attr(data-id)').get(),
                'title': item.css('h3.title::text').get(),
            }
            yield product
            yield scrapy.Request(
                item.css('a::attr(href)').get(),
                callback=self.parse_detail
            )

еӨҡиҠӮзӮ№ Worker йғЁзҪІж¶үеҸҠеҮ дёӘе…ій”®й—®йўҳпјҡиҠӮзӮ№еҗҜеҠЁдёҺеҒңжӯўгҖҒд»»еҠЎеҲҶй…ҚгҖҒзҠ¶жҖҒзӣ‘жҺ§гҖҒж•…йҡңжҒўеӨҚгҖӮ

иҠӮзӮ№еҗҜеҠЁдёҺеҒңжӯўпјҡжҜҸдёӘ Worker иҠӮзӮ№е°ұжҳҜдёҖдёӘжҷ®йҖҡзҡ„ Scrapy иҝӣзЁӢпјҢйҖҡиҝҮ scrapy crawl distributed_product еҗҜеҠЁгҖӮеҒңжӯўж—¶зӣҙжҺҘ Ctrl+C жҲ– kill иҝӣзЁӢеҚіеҸҜгҖӮз”ұдәҺйҳҹеҲ—е’ҢеҺ»йҮҚйӣҶеҗҲйғҪеңЁ Redis дёӯпјҢWorker зҡ„еҗҜеҒңдёҚдјҡдёўеӨұж•°жҚ®гҖӮж–°дә§з”ҹзҡ„иҜ·жұӮе·Із»ҸеҶҷе…Ҙ Redis йҳҹеҲ—пјҢеҚідҪҝжүҖжңү Worker йғҪеҒңдәҶпјҢдёӢж¬ЎеҗҜеҠЁеҗҺиҝҳиғҪз»§з»ӯеӨ„зҗҶгҖӮ

д»»еҠЎеҲҶй…ҚпјҡScrapy-Redis еӨ©з„¶ж”ҜжҢҒиҙҹиҪҪеқҮиЎЎгҖӮеӨҡдёӘ Worker д»ҺеҗҢдёҖдёӘйҳҹеҲ—з«һдәүиҜ·жұӮпјҢеӨ„зҗҶйҖҹеәҰеҝ«зҡ„иҠӮзӮ№иҮӘз„¶дјҡеӨ„зҗҶжӣҙеӨҡиҜ·жұӮгҖӮдёҚйңҖиҰҒжүӢеҠЁеҲҶй…Қд»»еҠЎгҖӮ

зҠ¶жҖҒзӣ‘жҺ§пјҡйҖҡиҝҮ Redis еҸҜд»Ҙе®һж—¶жҹҘзңӢйҳҹеҲ—зҠ¶жҖҒпјҡ

import redis

def monitor_queue(redis_url):
    r = redis.from_url(redis_url)
    queue_len = r.llen('product:requests')
    dupefilter_size = r.scard('product:dupefilter')
    print(f'еҫ…еӨ„зҗҶиҜ·жұӮ: {queue_len}')
    print(f'е·ІеҺ»йҮҚURL: {dupefilter_size}')

еӨҡиҠӮзӮ№йғЁзҪІж—¶иҝҳжңүдёҖдёӘе®№жҳ“иў«еҝҪи§Ҷзҡ„й—®йўҳпјҡеҗ„иҠӮзӮ№зҡ„д»Јз ҒзүҲжң¬еҝ…йЎ»дёҖиҮҙгҖӮеҰӮжһңжҹҗдёӘиҠӮзӮ№зҡ„и§ЈжһҗйҖ»иҫ‘жӣҙж–°дәҶдҪҶе…¶д»–иҠӮзӮ№жІЎжңүжӣҙж–°пјҢдјҡеҜјиҮҙж•°жҚ®ж јејҸдёҚдёҖиҮҙгҖӮе»әи®®дҪҝз”Ё Docker й•ңеғҸжқҘдҝқиҜҒеҗ„иҠӮзӮ№зҺҜеўғдёҖиҮҙпјҢй…ҚеҗҲ CI/CD жөҒж°ҙзәҝе®һзҺ°иҮӘеҠЁеҢ–йғЁзҪІгҖӮ

8.4.3 Scrapyd йғЁзҪІдёҺз®ЎзҗҶвҖӢ

жүӢеҠЁеңЁжҜҸеҸ°жңҚеҠЎеҷЁдёҠеҗҜеҠЁ Scrapy иҝӣзЁӢжҳҫз„¶дёҚеӨҹдјҳйӣ…гҖӮScrapyd жҳҜдёҖдёӘдё“й—Ёз”ЁдәҺйғЁзҪІе’ҢиҝҗиЎҢ Scrapy зҲ¬иҷ«зҡ„жңҚеҠЎпјҢжҸҗдҫӣдәҶ REST API жқҘз®ЎзҗҶзҲ¬иҷ«д»»еҠЎгҖӮ

Scrapyd зҡ„жһ¶жһ„пјҡScrapyd дҪңдёәдёҖдёӘзӢ¬з«ӢжңҚеҠЎиҝҗиЎҢеңЁжҜҸдёӘ Worker иҠӮзӮ№дёҠгҖӮе®ғжҺҘж”¶ HTTP иҜ·жұӮжқҘеҗҜеҠЁгҖҒеҒңжӯўгҖҒжҹҘиҜўзҲ¬иҷ«д»»еҠЎгҖӮзҲ¬иҷ«д»ҘеӯҗиҝӣзЁӢж–№ејҸиҝҗиЎҢпјҢScrapyd иҙҹиҙЈзӣ‘жҺ§иҝӣзЁӢзҠ¶жҖҒе№¶ж”¶йӣҶж—Ҙеҝ—гҖӮ

е…ҲеңЁ Worker иҠӮзӮ№дёҠе®үиЈ…е’ҢеҗҜеҠЁ Scrapydпјҡ

з„¶еҗҺз”Ё scrapyd-client жү“еҢ…дёҠдј йЎ№зӣ®пјҡ

scrapyd-deploy дјҡжҠҠйЎ№зӣ®жү“еҢ…жҲҗ egg ж–Үд»¶пјҢйҖҡиҝҮ HTTP дёҠдј еҲ° Scrapyd жңҚеҠЎеҷЁгҖӮдҪ йңҖиҰҒеңЁйЎ№зӣ®ж №зӣ®еҪ•зҡ„ scrapy.cfg дёӯй…ҚзҪ®зӣ®ж ҮжңҚеҠЎеҷЁдҝЎжҒҜпјҡ

  -d project=myproject \

еҒңжӯўд»»еҠЎпјҡ

иҝҷйҮҢжңүдёҖдёӘе®һжҲҳиё©еқ‘з»ҸйӘҢпјҡScrapyd й»ҳи®Өз”Ё pickle еәҸеҲ—еҢ–ж–№ејҸдҝқеӯҳд»»еҠЎзҠ¶жҖҒпјҢеҰӮжһңдҪ зҡ„ Scrapy зүҲжң¬е’Ң Scrapyd зүҲжң¬дёҚеҢ№й…ҚпјҢеҸҜиғҪдјҡеҮәзҺ°еәҸеҲ—еҢ–е…је®№жҖ§й—®йўҳгҖӮе»әи®®еңЁжүҖжңүиҠӮзӮ№дёҠдҪҝз”ЁзӣёеҗҢзүҲжң¬зҡ„ Scrapy е’Ң ScrapydпјҢжҲ–иҖ…з”Ё Docker з»ҹдёҖзҺҜеўғгҖӮ

еҸҰдёҖдёӘеқ‘пјҡScrapyd зҡ„ж—Ҙеҝ—й»ҳи®ӨеҶҷеҲ°зЈҒзӣҳж–Үд»¶пјҢй•ҝж—¶й—ҙиҝҗиЎҢдјҡеҚ з”ЁеӨ§йҮҸзЈҒзӣҳз©әй—ҙгҖӮйңҖиҰҒеңЁ scrapyd.conf дёӯй…ҚзҪ®ж—Ҙеҝ—иҪ®иҪ¬пјҡ

еҪ“ Worker иҠӮзӮ№и¶…иҝҮ 5 дёӘд»ҘдёҠж—¶пјҢжүӢеҠЁз®ЎзҗҶжҜҸдёӘиҠӮзӮ№зҡ„ Scrapyd е°ұеҸҳеҫ—еҫҲз—ӣиӢҰдәҶгҖӮдҪ йңҖиҰҒдёҖдёӘз»ҹдёҖзҡ„з®ЎзҗҶз•ҢйқўжқҘз®ЎзҗҶжүҖжңүиҠӮзӮ№зҡ„зҲ¬иҷ«д»»еҠЎгҖӮиҝҷйҮҢд»Ӣз»ҚдёӨдёӘдё»жөҒзҡ„ејҖжәҗж–№жЎҲпјҡGerapy е’Ң CrawlabгҖӮ

    def __init__(self, redis_client):
        self.redis = redis_client

            return item
        else:
